        // Функция хэширования строки для получения стабильного индекса
        const getHash = (str) => {
            let hash = 0;
            for (let i = 0; i < str.length; i++) {
                hash = str.charCodeAt(i) + ((hash << 5) - hash);
            }
            return Math.abs(hash);
        };

        /**
         * Улучшенная логика цветов:
         * 1. Получаем список уникальных упражнений из имеющихся данных.
         * 2. Упражнения до 10 индекса получают Желтый или Красный.
         * 3. Упражнения после 10 индекса получают Белый.
         * 4. Привязка идет по имени (хэш), но с учетом порядка появления в базе.
         */
        const getExerciseColor = (name, allExercises) => {
            // Находим индекс этого упражнения в общем списке (отсортированном для стабильности)
            const sortedExercises = [...allExercises].sort();
            const index = sortedExercises.indexOf(name);
            const hash = getHash(name);

            // Если упражнение входит в первую десятку
            if (index < 10) {
                // Чередуем желтый и красный на основе хэша
                const isYellow = hash % 2 === 0;
                if (isYellow) {
                    const lightness = 45 + (hash % 20); // 45-65%
                    return `hsl(50, 100%, ${lightness}%)`; // Желтые тона
                } else {
                    const lightness = 45 + (hash % 20); // 45-65%
                    return `hsl(0, 85%, ${lightness}%)`; // Красные тона
                }
            } else {
                // Для всех упражнений после 10-го — оттенки белого/светло-серого
                const lightness = 85 + (hash % 12); // 85-97%
                return `hsl(0, 0%, ${lightness}%)`;
            }
        };

        function renderChart() {
            const ctx = document.getElementById('progressChart').getContext('2d');
            if (chartInstance) chartInstance.destroy();
            if (workouts.length === 0) return;

            const period = currentPeriod;
            let filteredWorkouts = [...workouts];

            if (period !== 'all') {
                const daysLimit = parseInt(period);
                const cutoffDate = new Date();
                cutoffDate.setDate(cutoffDate.getDate() - daysLimit);
                filteredWorkouts = workouts.filter(w => new Date(w.date) >= cutoffDate);
            }

            const sorted = filteredWorkouts.sort((a, b) => new Date(a.date) - new Date(b.date));
            const dates = [...new Set(sorted.map(w => w.date))].sort();
            
            // Список всех уникальных упражнений (для логики цветов)

            const activeExercises = [...new Set(sorted.map(w => w.exercise))];

        function renderHistory() {
            const list = document.getElementById('history-list');
            const filtered = workouts.filter(w => (w.exercise || '').toUpperCase().includes(searchTerm));
            list.innerHTML = filtered.map(w => `
                <tr class="hover:bg-white/[0.02] transition-colors">
                    <td class="sm:p-4 text-slate-400 table-font-sync text-[11px] sm:text-xs" data-label="Дата">${w.date}</td>
                    <td class="sm:p-4 text-white table-font-sync text-xs sm:text-sm" data-label="Упр.">${w.exercise}</td>
                    <td class="sm:p-4" data-label="Вес">
                        <span class="bg-brandYellow/10 text-brandYellow px-3 py-1 rounded-lg table-font-sync text-[11px] border border-brandYellow/20">
                            ${w.result} КГ
                        </span>
                    </td>
                    <td class="sm:p-4 text-slate-400 text-[10px] sm:text-xs italic leading-tight" data-label="Заметка">${w.comment || '—'}</td>
                    <td class="sm:p-4 text-right" data-label="Действие">
                        <button onclick="deleteWorkout('${w.id}')" class="p-2 text-red-500/50 hover:text-red-500 transition-colors">
                            <i data-lucide="trash-2" class="w-4 h-4 ml-auto"></i>
                        </button>
                    </td>
                </tr>
            `).join('');
            lucide.createIcons();
        }

        window.deleteWorkout = async (id) => {
            try {
                const { error } = await supabaseClient.from('workouts').delete().eq('id', id);
                if (error) throw error;
                showToast("Запись удалена");
                await fetchWorkouts();
            } catch (e) { showToast("Ошибка удаления", "error"); }
        };

        function initCustomSuggestions() {
            const input = document.getElementById('exercise');
            const list = document.getElementById('exercise-suggestions');
            input.addEventListener('input', () => {
                const val = input.value.toUpperCase();
                const unique = [...new Set(workouts.map(w => w.exercise.toUpperCase()))].filter(ex => ex.includes(val)).sort();
                if (val && unique.length) {
                    list.innerHTML = unique.map(ex => `<div class="suggestion-item">${ex}</div>`).join('');
                    list.style.display = 'block';
                } else list.style.display = 'none';
            });
            list.addEventListener('click', (e) => {
                if (e.target.classList.contains('suggestion-item')) {
                    input.value = e.target.textContent;
                    list.style.display = 'none';
                }
            });
            document.addEventListener('click', (e) => { if (!input.contains(e.target)) list.style.display = 'none'; });
        }

        document.getElementById('search-input').addEventListener('input', (e) => {
            searchTerm = e.target.value.toUpperCase();
            renderHistory();
        });

        document.getElementById('workout-form').addEventListener('submit', async (e) => {
            e.preventDefault();
            const btn = document.getElementById('submit-btn');
            btn.disabled = true;
            const payload = {
                client_id: activeClientId,
                date: document.getElementById('date').value,
                exercise: document.getElementById('exercise').value.trim().toUpperCase(),
                result: parseFloat(document.getElementById('result').value),
                comment: document.getElementById('comment').value.trim()
            };
            try {
                const { error } = await supabaseClient.from('workouts').insert([payload]);
                if (error) throw error;
                showToast("Данные сохранены");
                document.getElementById('exercise').value = '';
                document.getElementById('result').value = '';
                document.getElementById('comment').value = '';
                document.getElementById('char-count').textContent = "0 / 100";
                await fetchWorkouts();
            } catch (err) { showToast("Ошибка сохранения", "error"); }
            finally { btn.disabled = false; }
        });
